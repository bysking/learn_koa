          const start = this.lease_range[0].getTime() - 1 * 24 * 60 * 60 * 1000 // 租期开始时间
          const end = this.lease_range[1].getTime() > new Date().getTime() ? new Date().getTime() : this.lease_range[1].getTime() // 租期结束时间和现在时间对比选择最小的那个因为当前时间之后没有产生消费记录
          const current = date.getTime() // 传入的时间
          if (start <= current && current <= end) {

 created () {
    // 租约时间范围
    // this.lease_range[0] = new Date('2019-3-1')
    // this.lease_range[1] = new Date('2019-9-10')
    this.params.start_time = this.lease_range[0] // 根据租约时间赋初值保存下来，下面会使用到租约时间来计算时间
    this.params.end_time = this.lease_range[1]

    this.params.end_time = new Date(this.lease_range[1]).getTime() > new Date().getTime() // 时间选择器的结束时间： 如果租约结束时间在当前时间之后就选当前时间zuo'wei结束时间，否则选择租约结束时间作为选择器的结束时间
      ? endOfDay(new Date())
      : endOfDay(this.lease_range[1]) // 租期结束时间

    const timeBefore90 = dayjs(new Date(this.params.end_time)).subtract(90, 'd').valueOf() // 结束时间的前90天 默认可让用户选择的时间范围90天

    this.params.start_time = timeBefore90 > this.params.start_time // 结束时间减去90天和租约开始时间比较，如果减去90天后还在租期范围内就选‘减去90天’对应的值，否则判断租约开始时间在今天之后的话，就只默认让今天作为开始，相应的结束时间也是今天，因为后面的天数未产生消费，不能赋初值
      ? timeBefore90
      : (new Date(this.params.start_time).getTime() > new Date().getTime()
        ? new Date()
        : this.params.start_time) // 时间选择器默认值赋值

    this.params.date = [dayjs(this.params.start_time).format('YYYY-MM-DD'), dayjs(this.params.end_time).format('YYYY-MM-DD')] // 时间选择器赋初值
