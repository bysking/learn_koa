import Vue from 'vue'

let vm = new Vue()
let toast

/**
  * @param {String} title 标题
  * @param {String} placeholder Input
  * @param {String} value 输入框初始值
  * @param {Regexp} regexp 输入框正则
  * @param {Funtion} cb 回调
*/
function showPromptDialog (title, placeholder, value, regexp, cb) {
  vm.$createDialog({
    type: 'prompt',
    title,
    prompt: {
      value,
      placeholder
    },
    onConfirm: (e, promptValue) => {
      if (!regexp.test(promptValue)) {
        showErr(placeholder)
        return
      }
      cb(promptValue)
    }
  }).show()
}

/**
  * @param {String} title 标题
  * @param {String} placeholder Input
  * @param {String} value 输入框初始值
  * @param {Regexp} regexp 输入框正则
  * @param {Funtion} cb 回调
*/
function showPromptDialogBtn (title, content, cb) {
  vm.$createDialog({
    type: 'confirm',
    title,
    content,
    confirmBtn: {
      text: '确定按钮',
      active: true,
      disabled: false,
      href: 'javascript:;'
    },
    cancelBtn: {
      text: '取消按钮',
      active: false,
      disabled: false,
      href: 'javascript:;'
    },
    onConfirm: (e, promptValue) => {
      let data = true
      cb(data)
    },
    onCancel: (e, promptValue) => {
      let data = false
      cb(data)
    }
  }).show()
}

/**
  * 显示toast
  * @param {String} title 标题
  * @param {String} content 描述
  * @param {Funtion} cb 回调
*/
function showConfirmDialog (title, content, cb) {
  vm.$createDialog({
    type: 'confirm',
    title,
    content,
    onConfirm: () => {
      cb()
    }
  }).show()
}
/**
  * 显示toast
  * @param {String} title 标题
  * @param {String} content 描述
*/
function showAlertDialog (title, content) {
  vm.$createDialog({
    type: 'alert',
    title,
    content
  }).show()
}
/**
  * 显示toast
  * @param {String} msg 信息
  * @param {String} mask 是否显示模态层
  * @param {String} time 时间，为0需要手动关闭
  * @param {String} type 类型（loading/correct/error/warn/txt）
  */
function baseToast (msg, mask = false, time = 1500, type = 'txt') {
  toast = vm.$createToast({
    type: type,
    txt: msg,
    mask: mask,
    time: time
  })
  toast.show()
}

/**
  * 显示toast
  * @param {String} msg 信息
  */
function showToast (msg) {
  baseToast(msg)
}

/**
  * 显示警告
  * @param {String} msg 信息
  */
function showWarn (msg) {
  baseToast(msg, false, 1500, 'warn')
}

/**
  * 显示错误
  * @param {String} msg 信息
  */
function showErr (msg) {
  baseToast(msg, false, 1500, 'error')
}

/**
  * 显示正确
  * @param {String} msg 信息
  */
function showCorrect (msg, time = 1500) {
  baseToast(msg, false, time, 'correct')
}

/**
  * 显示loading
  * @param {String} msg 信息
  */
function showLoading (msg = '加载中...') {
  baseToast(msg, true, 0, 'loading')
}

// 隐藏toast
function hideToast () {
  if (toast) {
    toast.hide()
  }
}

// 拨打电话

export const moveDialogPoition = (reset = false) => {
  if (reset) {
    setTimeout(() => {
      document.body.classList.remove('prompt-dialog-focus')
    }, 200)
  } else {
    document.body.classList.add('prompt-dialog-focus')
  }
}

  showErr,
  showCorrect,
  showLoading,
  hideToast,
